#### 试题D：最短路

> dijkstra，[算法模板](https://blog.csdn.net/YMWM_/article/details/115497375)。

```c++
#include<iostream>

using namespace std;

int ver[20][20];  // 顶点 ver[i][j]表示i j之间的距离
int dist[20];  // 距离 dist[i]表示i到起点的最短距离
bool known[20];  // 标记 是否被处理过
int n = 19;  // 题目顶点数

// dijkstra
int Dijkstra() {
	memset(dist, 0x3f, sizeof(dist));  // 距离初始化为inf
	dist[1] = 0;  // 第一个顶点距离为0 自己到自己的距离

	// 遍历所有顶点
	for (int i = 0; i < n; ++i) {
		int t = -1;
		// 遍历寻找距离最近的下一个顶点
		for (int j = 1; j <= n; ++j) {
			if (!known[j] && (t == -1 || dist[t] > dist[j]))
				t = j;
		}

		// 更新最小距离
		for (int j = 1; j <= n; ++j) {
			dist[j] = min(dist[j], dist[t] + ver[t][j]);
		}

		// 更新标记
		known[t] = true;
	}
	
	return dist[19];
}

// 初始化边
void Add(int a, int b, int c) {
	ver[a][b] = ver[b][a] = c;
}

// 单源最短路径问题：给定一个加权图和一个特定顶点作为输入
// 找到从输入到图中任意一个顶点的最短加权路径
```

#### [试题G：外卖店优先级](https://blog.csdn.net/qq_59700927/article/details/123131003)

> 题目怎么说就怎么写，pair，按照时间排序。

```c++
#include<iostream>
#include<algorithm>

using namespace std;

const int maxn = 100010;
int n, m, t;  // 输入
int last[maxn];  // 记录上一次出现订单的时间
int v[maxn];  // 表示某家点的分数
pair<int, int> order[maxn];  // 存放订单
bool res[maxn];  // 判断某家店是否在优先缓存中

int main() {
	//初始化
	cin >> n >> m >> t;
	for (int i = 1; i <= m; ++i) {
		cin >> order[i].first >> order[i].second;
	}
	for (int i = 1; i <= max(n, m); ++i) {
		last[i] = 0;
		v[i] = 0;
		res[i] = false;
	}

	// 按照订单时间升序
	sort(order + 1, order + 1 + m);
	// 处理数据 依次遍历每个订单
	for (int i = 1; i <= m; ++i) {
		int name = order[i].second;  // 获得订单的店名
		int ti = order[i].first;  // 获得订单的时间
		int ans = 1;  // 订单数量
		// 同时刻同家店订单处理
		while ((name = order[i + 1].second) && (ti == order[i + 1].first)) {
			++i;
			++ans;
		}
		if (ti - last[name] > 0) 
			// 如果本次获得订单的时间与上次获得订单的时间有间隔 优先分数要降
			v[name] -= (ti - last[name] - 1);  
		last[name] = ti;  // 更新上次获得订单的时间
		if (v[name] < 0)
			v[name] = 0;  // 最低分为0
		if (v[name] <= 3)
			res[name] = false;  // 低于三分移出优先缓存
		v[name] += (ans * 2);  // 订单数*每单加的分数
		if (v[name] > 5)
			res[name] = true;  // 如果大于5分 放进优先缓存中
	}

	// 判断t时刻的优先缓存中的数量
	for (int i = 1; i <= n; ++i) {
		if (t > last[i])
			v[i] -= (t - last[i]);  // 如果t时刻比上次出现订单的时刻大 要减分
		// 更新优先级
		if (v[i] <= 3)
			res[i] = false;  
		if (v[i] > 5)
			res[i] = true;
	}

	// 输出
	int cnt = 0;
	for (int i = 1; i <= n; ++i) {
		if (res[i])
			++cnt;
	}
	cout << cnt;
	return 0;
}
```

#### 试题D：数的分解

```c++
#include<iostream>

using namespace std;

// 判断是否包含2或4(常见判断是否包含某个数字)
bool NotInclude24(int a) {
	while (a) {
		if (a % 10 == 2 || a % 10 == 4) 
			return false;
		else
			a /= 10;
	}
	return true;
}

int main() {
	int count = 0;
	for (int i = 1; i <= 2019; ++i) {
		for (int j = 1; j <= 2019; ++j) {
			for (int k = 1; k <= 2019; ++k) {
				if (i == j || i == k || j == k)
					continue;  // 数字各不相同
				if (i + j + k == 2019 && NotInclude24(i)
					&& NotInclude24(j) && NotInclude24(k))
					++count;

			}
		}
	}


	cout << count / 6;
	return 0;
}
```

#### 试题G：完全二叉树权值

> 完全二叉树不等于满二叉树，不要混淆！还有就是数字范围的时候，很大直接long long。

```c++
#include<iostream>
#include<cmath>

using namespace std;

const int n = 101000;
const int inf = 1e9 + 7;
typedef long long ll;
int arr[n];

int main() {
	ll n, deep, maxsum = -inf, ans;
	// 输入
	cin >> n; 
	for (int i = 1; i <= n; ++i) {
		cin >> arr[i];
	}
	
	// 计算深度
	for (int i = 1; ; ++i) {
		if (pow(2, i) - 1 >= n) {
			deep = i;
			break;
		}
			
	}

	for (int i = 1; i <= deep; ++i) {
		ll sum = 0;
		for (int j = pow(2, i - 1); j <= pow(2, i) - 1 && j <= n; ++j) {
			sum += arr[j];  // 计算当前深度的权值和
		}
		if (sum > maxsum) {
			maxsum = sum;  // 更新最大值
			ans = i;  // 因为输出层数最低的最大值 所以依次遍历即可
		}
	}

	cout << ans;
	return 0;
}
```

#### [试题J：组合数问题](https://www.cnblogs.com/mikku39/p/14608578.html)

> 看了题解挺难的，不知道为啥暴力搜索结果会不一样。

```c++
#include<iostream>
#include<vector>
#include<cmath>

using namespace std;

typedef long long ll;
const int mod = 1e9 + 7;
ll t, k;
ll arr[100005][2];

// 计算组合数的函数
int C(ll a, ll b) {
	ll ans = 1;
	for (ll i = 0; i < b; ++i) {
		ans *= (a - i) / (b - i);
	}
	return ans;
}

int main() {
	// 输入
	cin >> t >> k;
	for (int i = 0; i < t; ++i) {
		for (int j = 0; j < 2; ++j) {
			cin >> arr[i][j];
		}
	}
	
	vector<ll> res;
	// 依次遍历每组询问
	for (int num = 0; num < t; ++num) {
		int ans = 0;
		for (ll i = 1; i <= arr[num][0]; ++i) {
			for (ll j = 0; j <= min(i, arr[num][1]); ++j) {
				if (C(i, j) % k == 0)
					++ans;
			}
		}
		res.push_back(ans);
	}

	for (ll i = 0; i < res.size(); ++i) {
		cout << res[i] % mod << endl;
	}
	return 0;
}
```
